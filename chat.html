<button onclick="send()">Сохранить чат</button>


<p id="status"></p>


<script>

const WEBHOOK = 'ТВОЙ_WEBHOOK';


function getParam(name) {

  return new URL(window.location.href).searchParams.get(name) || '';

}


document.addEventListener('DOMContentLoaded', () => {

  document.getElementById('manager').value = getParam('manager');

  document.getElementById('phone').value = getParam('phone');

});


async function send() {

  const data = {

    type: 'chat', // ВАЖНО для гибрида

    manager: document.getElementById('manager').value,

    phone: document.getElementById('phone').value.replace(/\D/g, ''),

    comment: document.getElementById('comment').value,

    result: document.getElementById('result').value

  };


  await fetch(WEBHOOK, {

    method: 'POST',

    body: JSON.stringify(data)

  });


  document.getElementById('status').innerText = 'Сохранено';

}

</script>
